Juniper JunOS

protocols {
    bgp {
        group AS34388-TR-IX-RS1 {
            type external;
            description "TR-IX Route Server 1";
            remove-private;
            peer-as 34388;
            neighbor 185.1.158.254 {
                import TR-IX-IN-v4;
                family inet {
                    unicast {
                        prefix-limit {
                            maximum 160000;
                            teardown 75 idle-timeout 60;
                        }
                    }
                }
                authentication-key PASSWORD;
                export TR-IX-OUT-v4;
            }
            neighbor 2001:7f8:ea::3:4388:1 {
                import TR-IX-IN-v6;
